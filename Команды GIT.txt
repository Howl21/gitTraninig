git init 				- инициализировать репозиторий;
git add 				- добавить файл(ы) к отслеживанию;
git commit -m |-a |--amend		- сделать снимок файлов | git add . -> git commit -m (Только для modified!)| дополняет последний комит изменениями, не создавая новый. Можно изменить сообщение коммита;
git diff				- посмотреть изменения между последним коммитом и последним незафиксированным изменением;
git reset --soft|--mixed|--hard		- отмена изменений (отменяет коммит и оставляет в отслеживаемой зоне | -//- и оставляет в неотслеживаемой зоне | удаляет изменения)
git checkout				- перемещения между коммитами, версиями файлов, ветками;
git clean -n|-f				- показывает файлы к удалению | удаляет untracked файлы;

//////////////////////////////  РАБОТА С УДАЛЕННЫМИ РЕПОЗИТОРИЯМИ   ////////////////////////////

git remote -v|add НАЗВАНИЕ_РЕПОЗИТОРИЯ АДРЕС_РЕПОЗИТОРИЯ|remove НАЗВАНИЕ_РЕПОЗИТОРИЯ		- просмотр списка репозиториев| добавление нового репозитория| удаление репозитория;
git push НАЗВАНИЕ_РЕПОЗИТОРИЯ ВЕТКА								- отправляет данные на удаленный репозиторий (Название по умолчанию origin);
git pull НАЗВАНИЕ_РЕПОЗИТОРИЯ ВЕТКА| --delete НАЗВАНИЕ_РЕПОЗИТОРИЯ ВЕТКА			- получение данных из удаленного репозитория| удаление ветки из репозитория;
git clone АДРЕС_РЕПОЗИТОРИЯ									- клонирование репозитория;
											
//////////////////////////////  РАБОТА С ВЕТКАМИ   ////////////////////////////

git branch |НАЗВАНИЕ_ВЕТКИ | -d НАЗВАНИЕ_ВЕТКИ |-r						- посмотреть текущую ветку| создание новой ветки| удаление ветки локально| список удаленных веток;
git checkout НАЗВАНИЕ_ВЕТКИ 									- переключение между ветками;
git merge НАЗВАНИЕ_ВЕТКИ 									- слияние веток (merge вызываем с той ветки, в которую хотим слить изменения);
git fetch 											- получение данных в удаленную ветку (origin/НАЗВАНИЕ_ВЕТКИ)
git remote show НАЗВАНИЕ_РЕПОЗИТОРИЯ								- подробная информация о ветках
